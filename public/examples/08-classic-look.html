#classic-range {
    --track-height: 4px;
    --thumb-size: 24px;
}
#classic-range::part(track) {
    background: #444;
    border: 1px solid #111;
    border-radius: 0;
    box-shadow: inset 0 1px 2px rgba(0, 0, 0, 0.5);
}
#classic-range::part(segment-2) {
    background: #3b82f6;
}
#classic-range::part(thumb) {
    anchor-name: var(--anchor-name);
    background: linear-gradient(to bottom, #f9fafb, #d1d5db);
    border: 1px solid #6b7280;
    border-radius: 3px;
    width: 14px;
    height: var(--thumb-size);
    box-shadow: 0 1px 2px rgba(0, 0, 0, 0.5);
}
#classic-range::part(thumb-1) { --anchor-name: --classic-thumb-1; }
#classic-range::part(thumb-2) { --anchor-name: --classic-thumb-2; }
